class Ques(models.Model):

# 单选题

sections = models.ForeignKey(Section, verbose_name='学段', on_delete=models.CASCADE, blank=True, null=True, )
subjects = models.ForeignKey(Subject, verbose_name='学科', on_delete=models.CASCADE, blank=True, null=True, )
questype = models.ForeignKey(QuesType, verbose_name="题型", on_delete=models.CASCADE, blank=True, null=True,)
level = models.ForeignKey(Level, verbose_name="难易度", on_delete=models.CASCADE, blank=True, null=True,)
knowledge = models.ManyToManyField(MPTTLedge, verbose_name="知识点",  blank=True, null=True,)

score = models.FloatField('分值',  blank=True, default=0.0)
author = models.CharField('作者', null=True, blank=True, max_length=50)
add_time = models.DateTimeField("添加时间", default=datetime.now, )
update_time = models.DateTimeField("更新时间", default=datetime.now, )

content=RichTextField(blank=True,null=True,verbose_name="题干")
answer=RichTextField(blank=True,null=True,verbose_name="答案")
analysis = RichTextField(blank=True,null=True,verbose_name="解析")
quote_times = models.IntegerField("使用次数", default=0)
fav_nums = models.IntegerField("收藏人数", default=0)
click_nums = models.IntegerField("点击数", default=0)
tag = models.CharField('试题标签', default='', max_length=10,null=True, blank=True, )

class Meta:
    verbose_name = "试题"
    verbose_name_plural = verbose_name

def __str__(self):
    return self.author
